class Solution {
    public void sortColors(int[] nums) {
        int p0 = 0, p1 = 0;
        for(int i = 0; i < nums.length; i++){
            if(nums[i] == 0){
                change(nums,i,p0);
                if(p0 < p1){
                    change(nums,i,p1);
                }
                p0++;
                p1++;
            }else if(nums[i] == 1){
                change(nums,i,p1);
                p1++;
            }
            
        }
    }
    void change(int[] nums, int a, int b){
        int temp = nums[a];
        nums[a] = nums[b];
        nums[b] = temp;
    }
}

            swap(nums, left++, right--);

class Solution {
    public int findDuplicate(int[] nums) {
        int fast = 0, slow = 0;
        do{
            fast=nums[nums[fast]];
            slow = nums[slow];
        }while(fast!=slow);
        slow = 0;
        while(fast!=slow){
            fast=nums[fast];
            slow = nums[slow];
        }
        return fast;
    }
}
